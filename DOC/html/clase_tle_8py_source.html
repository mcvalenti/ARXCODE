00031         REVISAR!!! 
00032         ---------------
00033         outputs
00034             alpha, delta: angulos Geodesicos (float) [grados]

00036         deg=np.divide(180,np.pi)
00037         r_mod=np.sqrt(self.r[0]*self.r[0]+self.r[1]*self.r[1]+self.r[2]*self.r[2])
00038         l=np.divide(self.r[0],r_mod)
00039         m=np.divide(self.r[1],r_mod)
00040         n=np.divide(self.r[2],r_mod)
00041         self.delta=np.arcsin(n)
00042         if m>0:
00043             self.alpha = np.arccos(np.divide(l,np.cos(self.delta)))
00044         else:
00045             self.alpha = 2*np.pi-np.arccos(np.divide(l,np.cos(self.delta)))
00046
